2)Set java home varible in Bash rc file of linux 
- ls /usr/lib/jvm/
- nano ~/.bashrc
- export JAVA_HOME=/usr/lib/jvm/java-11-openjdk-amd64
- export PATH=$PATH:$JAVA_HOME/bin
- source ~/.bashrc
- echo $JAVA_HOME
- echo $PATH
